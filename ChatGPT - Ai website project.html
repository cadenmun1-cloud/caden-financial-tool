🚀 How to Add REAL Stock Data to Your Website

Instead of manually typing returns, your site can pull real data automatically (like Apple, Tesla, etc.).

This version:

  • Fetches real stock prices

  • Converts them into returns

  • Calculates everything automatically

HTML
<script>

async function getStockData(ticker) {
  let url = `https://query1.finance.yahoo.com/v8/finance/chart/${ticker}?range=1mo&interval=1d`;
  
  let response = await fetch(url);
  let data = await response.json();

  let prices = data.chart.result[0].indicators.quote[0].close;

  return prices;
}

function calculateReturns(prices) {
  let returns = [];
  for (let i = 1; i < prices.length; i++) {
    returns.push(((prices[i] - prices[i-1]) / prices[i-1]) * 100);
  }
  return returns;
}

function average(data) {
  return data.reduce((a,b)=>a+b,0) / data.length;
}

function stdDev(data, avg) {
  let variance = data.reduce((sum,x)=> sum + Math.pow(x-avg,2),0) / data.length;
  return Math.sqrt(variance);
}

function calcBeta(stock, market) {
  let avgStock = average(stock);
  let avgMarket = average(market);

  let covariance = 0;
  let marketVariance = 0;

  for (let i = 0; i < stock.length; i++) {
    covariance += (stock[i]-avgStock)*(market[i]-avgMarket);
    marketVariance += Math.pow(market[i]-avgMarket,2);
  }

  return covariance / marketVariance;
}

async function calculateReal() {

  let stockTicker = document.getElementById("stockTicker").value;
  let marketTicker = "SPY"; // S&P 500 proxy
  let rf = Number(document.getElementById("rf").value);

  let stockPrices = await getStockData(stockTicker);
  let marketPrices = await getStockData(marketTicker);

  let stockReturns = calculateReturns(stockPrices);
  let marketReturns = calculateReturns(marketPrices);

  let avg = average(stockReturns);
  let vol = stdDev(stockReturns, avg);
  let beta = calcBeta(stockReturns, marketReturns);
  let sharpe = (avg - rf) / vol;

  document.getElementById("avg").innerText = avg.toFixed(2) + "%";
  document.getElementById("vol").innerText = vol.toFixed(2);
  document.getElementById("beta").innerText = beta.toFixed(2);
  document.getElementById("sharpe").innerText = sharpe.toFixed(2);
}

</script>
